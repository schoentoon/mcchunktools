15 /*
16  * A buffer is 'unlimited' storage for raw data. As long as buffer_append is
17  * used to add data, it will automatically resize to make room. To read the
18  * data, just access `data' directly.
19  */
20 struct buffer {
21  unsigned char* data; /* You can access the buffer's raw bytes through this pointer */
22  size_t len; /* Only accesses in the interval [data, data + len) are defined */
23  size_t cap; /* Internal use. The allocated size of the buffer. */
24 };
